	Class<?> paramType = parameter.getParameterType();

		ServletRequest.class.isAssignableFrom(paramType) ||
		MultipartRequest.class.isAssignableFrom(paramType) ||
		HttpSession.class.isAssignableFrom(paramType) ||

		Principal.class.isAssignableFrom(paramType) ||
		InputStream.class.isAssignableFrom(paramType) ||
		Reader.class.isAssignableFrom(paramType) ||
		HttpMethod.class == paramType ||
		Locale.class == paramType ||
		TimeZone.class == paramType ||

}

import org.springframework.http.server.ServerHttpResponse;
import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.servlet.mvc.method.annotation.ResponseBodyAdvice;

/**
 * еҜ№иҝ”еӣһз»“жһңеӨ„зҗҶпјҢз»ҹдёҖиҝ”еӣһж јејҸгҖӮ
 */
@Slf4j
@ControllerAdvice(basePackages = "com.example.demo2.v1")
public class ResultResponseHandler implements ResponseBodyAdvice<Object> {

    /**
     * жҳҜеҗҰеҜ№иҝ”еӣһз»“жһңиҝӣиЎҢеӨ„зҗҶ
     * @param methodParameter ж–№жі•зҡ„еҸӮж•°
     * @param aClass зұ»
     * @return true
     */
    @Override
    public boolean supports(MethodParameter methodParameter, Class<? extends HttpMessageConverter<?>> aClass) {
        return true;
    }

    @Override
    public Object beforeBodyWrite(Object o, MethodParameter methodParameter, MediaType mediaType, Class<? extends HttpMessageConverter<?>> aClass, ServerHttpRequest serverHttpRequest, ServerHttpResponse serverHttpResponse) {

        /**
         * ж—ҘжҠҘејӮеёёиҝ”еӣһеӨ„зҗҶ
         */
        if (o instanceof DiariesException) {
            return R.error(((DiariesException) o).getrHttpStatusEnum(), "1" + System.currentTimeMillis());
        }
        /**
         * йҖҡз”ЁејӮеёёиҝ”еӣһзұ»еһӢ
         */
        if (o instanceof Exception) {
            return R.error(RHttpStatusEnum.SERVER_ERROR);
        }
        /**
         * еӣ дёәеҰӮжһңиҝ”еӣһжҳҜstringзҡ„иҜқй»ҳи®Өдјҡи°ғз”Ёstringзҡ„еӨ„зҗҶеҷЁзӣҙжҺҘиҝ”еӣһпјҢжүҖд»ҘиҰҒиҝӣиЎҢеӨ„зҗҶ
         */
        if (o instanceof String) {
            return JsonUtil.obj2String(R.ok(o));
        }
        /**
         * жӯЈеёёиҝ”еӣһ
         */
        return R.ok(o);
    }
}
