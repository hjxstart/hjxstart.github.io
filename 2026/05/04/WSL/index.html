WSL

安装

开启 WSL 功能

# 以 管理员身份 运行 PowerShell 或命令提示符。输入以下命令并回车：
wsl --install

安装在线版

# 查看可用版本：
wsl --list --online
# 安装指定版本（例如 Debian）：
wsl --install -d Debian

安装离线包

# 1. 下载离线 Ubuntu Base 镜像
http://cdimage.ubuntu.com/ubuntu-base/releases/26.04/release/ubuntu-base-26.04-base-amd64.tar.gz
# 2. 准备安装目录
E:\WSL\Ubuntu
# 3. 使用 wsl --import 进行安装。语法：wsl --import <自定义名称> <安装位置> <镜像路径>
wsl --import Ubuntu-Base E:\WSL\Ubuntu E:\WSL\ubuntu-base-22.04-base-amd64.tar.gz
# 4. 启动并初始化系统
wsl -d Ubuntu-Base

使用

初始化配置

# ubuntu-base 是极简版。进入后你需要先更新源并安装基础工具：
apt update
apt install sudo vim networking-tools -y

基础管理命令

# 查看已安装状态
wsl --list --verbose # wsl -l -v

# 停止运行中的 WSL
wsl --terminate <实例名称>

# 关闭所有实例
wsl --shutdown

# 注销/彻底删除 (这会清空该系统内的所有数据)
wsl --unregister <实例名称>

kylin wsl

Kylin Desktop

# 0. ISO镜像中有 filesystem.squashfs 文件
casper/filesystem.squashfs

# 1. 创建挂载点和目标存放目录
sudo mkdir -p /mnt/temp

# 2. 挂载文件（假设你把文件放到了 E:\diyiso\）
# WSL 中 E 盘通常挂载在 /mnt/e/
sudo mount -t squashfs /mnt/e/diyiso/filesystem.squashfs /mnt/temp

# 3. 打包为 tar 文件（存回到 E 盘供之后导入）
# 注意：cd 进去打包可以确保 tar 包内的路径是从根开始的，没有多余的层级
cd /mnt/temp
sudo tar -cvf /mnt/e/diyiso/kd-2403-x86.tar .

# 4. 卸载挂载点
cd ~
sudo umount /mnt/temp

# 5. 导入新的麒麟系统
wsl --import kd-2403-x86 E:\WSL\KylinOS E:\diyiso\kd-2403-x86.tar

打包字体

# 将脚本保存为 kylin-fonts-create.sh 并赋予执行权限：
chmod +x kylin-fonts-create.sh
# 假设你的字体文件都存放在 E:\fonts（在 WSL 中对应 /mnt/e/fonts），执行以下命令：
./kylin-fonts-create.sh -n my-custom-fonts -f "/mnt/e/fonts"

更新镜像

# 确保环境已安装必要工具：
sudo apt update && sudo apt install rsync genisoimage md5sum -y
# 运行更新脚本
sudo bash kylin-iso.refresh.sh Kylin-Desktop-V10-SP1.iso
# 脚本自动化处理流程
1. 挂载与同步：脚本会自动将原始 ISO 挂载到 /mnt/peter，并将内容同步到工作目录 /mnt/peter-livecd。  
2. 更新校验和：自动重新计算所有文件的 MD5 值并覆盖 md5sum.txt，确保系统安装时的完整性检查能通过。  
3. 生成新镜像：调用 genisoimage 重新打包，并保留 UEFI 引导支持（使用 boot/grub/efi.img）。  
4. 输出结果：生成名为 Kylin-Desktop-V10-SP1-custom.iso 的新镜像，并赋予 777 权限。

下载其他架构的包

## query_links.sh
#!/bin/bash

# 1. 检查并添加架构
ensure_arch() {
    local arch=$1
    if ! dpkg --print-foreign-architectures | grep -q "$arch"; then
        echo "检测到尚未配置 $arch 架构，正在添加..."
        sudo dpkg --add-architecture "$arch"
        echo "正在更新软件包索引（需要一定时间）..."
        sudo apt update
    fi
}

echo "=== 麒麟 V10 软件包下载链接交互查询工具 ==="

read -p "请输入软件包名称 (例如 hplip): " PKG_NAME
[ -z "$PKG_NAME" ] && exit 1

echo -e "\n请选择目标架构:"
select ARCH in "amd64" "arm64" "loongarch64" "退出"; do
    [ "$ARCH" == "退出" ] && exit 0
    if [ -n "$ARCH" ]; then
        # 如果不是本机原生架构，检查是否需要执行 update
        if [ "$ARCH" != "$(dpkg --print-architecture)" ]; then
            ensure_arch "$ARCH"
        fi
        break
    fi
done

# 2. 获取版本
echo -e "\n正在从索引中检索 $PKG_NAME:$ARCH 的版本..."
mapfile -t VERSIONS < <(apt-cache madison "${PKG_NAME}:${ARCH}" 2>/dev/null | awk -F'|' '{print $2}' | sed 's/ //g')

if [ ${#VERSIONS[@]} -eq 0 ]; then
    echo "❌ 错误：在源中找不到 ${PKG_NAME}:${ARCH}"
    echo "提示：如果已经执行过 update 仍无法找到，请检查 /etc/apt/sources.list 是否包含该架构的源地址。"
    exit 1
fi

echo "请选择版本:"
select VER in "${VERSIONS[@]}" "退出"; do
    [ "$VER" == "退出" ] && exit 0
    if [ -n "$VER" ]; then break; fi
done

# 3. 输出链接
echo -e "\n--- 查询结果 ---"
# 仅提取该包本身的链接，过滤掉可能的依赖链接（按需调整）
RESULT=$(apt-get download --print-uris "${PKG_NAME}:${ARCH}=${VER}" 2>/dev/null | grep "^'")

if [ -z "$RESULT" ]; then
    echo "无法获取链接。"
else
    echo "$RESULT" | sed -n "s/^'\(.*\)'.*$/\1/p"
fi
